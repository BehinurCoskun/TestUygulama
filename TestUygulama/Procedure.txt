CREATE proc IslemleriListele 
@MalKodu varchar(30),
@BaslangicTarih int,
@BitisTarihi int
as
begin

Select 
 	ROW_NUMBER() Over(
		Order By Tarih
	) 'Sıra No',
  Case IslemTur
		When 0 Then 'Giriş'
		else 'Çıkış'
  End As 'Islem Tur',

  EvrakNo, CONVERT(VARCHAR(15), CAST(STI.Tarih - 2 AS datetime), 104) As Tarih,

  Case IslemTur
		When 0 then Miktar
		Else 0
  End As 'Giris Miktar',

  Case IslemTur
		When 1 then Miktar
		Else 0
  End As 'Cikis Miktar',

  Sum(Case 
		When IslemTur = 0 then Miktar
		Else -Miktar End) Over (Order by Tarih, ID) as 'Stok'

	from STI where Tarih >= CONVERT(datetime, @BaslangicTarih ,104)
					And Tarih <= CONVERT(datetime, @BitisTarihi ,104) 
					And MalKodu = @MalKodu

end
